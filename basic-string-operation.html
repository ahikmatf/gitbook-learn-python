String

astring = "My name is John Herlington, I am from Rusia"
print(len(astring)) #output 43

print(astring.count("n")) #output 4
print(astring.index("n")) #output 3

print(astring[3:8]) #output name
print(astring[0:8:3]) #output Mne

print(astring.upper()) #output MY NAME IS JOHN HERLINGTON, I AM FROM RUSIA
print(astring.lower()) #output my name is john herlington, i am from rusia

print(astring.startswith("My")) #output True
print(astring.endswith("Rusia")) #output False

print(astring.split("e")) #output ['My nam', ' is John H', 'rlington, I am from Rusia']
