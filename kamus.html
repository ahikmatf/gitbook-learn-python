phonebook = {}
phonebook["John"] = 123445
phonebook["Jack"] = 787878
phonebook["Green"] = 787658
print(phonebook) #output {'John': 123445, 'Jack': 787878, 'Green': 787658}

phonebook["Herlington"] = [12134, 797878, 67686]
print(phonebook) 
#output {'John': 123445, 'Jack': 787878, 'Green': 787658, 'Herlington': [12134, 797878, 67686]}
print(phonebook["Herlington"]) #print phonebook Herlington
#output [12134, 797878, 67686]
phonebook.clear() #clear semua list di phonebook

phonebook["Arnold"] = { "hp": 3435435, "facebook": "arnold.white", "instagram": "@arnold.white" }
print(phonebook["Arnold"]["facebook"]) #print arnold.white
